import { Card, CardGrid } from '@astrojs/starlight/components';

<CardGrid stagger>
  <Card title="Fully offline" icon="laptop">
    Powered by [sherpa-onnx](https://github.com/k2-fsa/sherpa-onnx). All inference runs on your machine — no internet required, no data leaves your device.
  </Card>
  <Card title="LLM post-processing" icon="translate">
    Connect any OpenAI-compatible API (Ollama, SiliconFlow, and more) to correct, reformat, or translate recognition results in real time.
  </Card>
  <Card title="Command mode" icon="pencil">
    Select text → hold the command key → speak your instruction → release. The selection is rewritten by the LLM instantly.
  </Card>
  <Card title="Scene management" icon="setting">
    Switch post-processing prompts on the fly with a hotkey. Each scene carries its own system prompt, provider, and model.
  </Card>
  <Card title="Multiple ASR providers" icon="list-format">
    Use the built-in local engine or plug in a cloud provider (OpenAI, ElevenLabs, Doubao, or your own script).
  </Card>
  <Card title="vinput CLI" icon="rocket">
    Manage models, scenes, LLM providers, and daemon state from the terminal — no manual config file editing needed.
  </Card>
</CardGrid>
